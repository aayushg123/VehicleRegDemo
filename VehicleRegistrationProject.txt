Order of table creation 

State-> Owner-> Vehicle->Mapper table (between owner and vehicle)


Vehicle Table
=========================
CREATE TABLE vehicle (
	vehicle_id INT NOT NULL AUTO_INCREMENT/ GENERATED BY DEFAULT AS IDENTITY,
	model varchar (255) NOT NULL,
	engine_no varchar (255) UNIQUE NOT NULL,
	regDate DATE,
	state varchar (100) NOT NULL,
	district varchar (255) NOT NULL,
	aadhar_card VARCHAR (12) NOT NULL,
	vehicle_regno varchar(15) NOT NULL PRIMARY KEY,
	challanNo INT DEFAULT 0,
	lastChallanPaid DATE,
	isArchived BIT
	FOREIGN KEY(aadhar_card) REFERENCES owner(aadhar_card)
);

State Table
=============

CREATE TABLE state (
	state varchar(255) NOT NULL,
	district varchar(255) NOT NULL,
	prefix varchar (10) NOT NULL,
	alphaseries varchar (4) DEFAULT 'A',
	next_number varchar(4) DEFAULT '0001'
);


Owner Table
=================
CREATE TABLE owner (
	owner_id INT NOT NULL AUTO_INCREMENT/ GENERATED BY DEFAULT AS IDENTITY,
	owner_name varchar (255) NOT NULL,
	dob date NOT NULL,
	aadhar_card VARCHAR(12) NOT NULL PRIMARY KEY,
	mobile_no BIGINT UNIQUE NOT NULL
);


Mapper Table
=============

Create TABLE mapper(
vehicle_regno varchar (15) NOT NULL,
mobile_no BIGINT NOT NULL
FOREIGN KEY(mobile_no) REFERENCES owner(mobile_no)
);
